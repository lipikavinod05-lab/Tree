int minDepth(struct TreeNode* root) {
    if (root == NULL)
        return 0;

   
    if (root->left == NULL)
        return 1 + minDepth(root->right);

    if (root->right == NULL)
        return 1 + minDepth(root->left);

   
    int lh = minDepth(root->left);
    int rh = minDepth(root->right);

    return (lh < rh ? lh : rh) + 1;
}
